int suma(int a, int b) {
    int resultado;
    resultado = a + b;
    return resultado;
}

int main() {
    int x, y, z;
    x = 5;
    y = 10;
    
    if (x < y) {
        z = suma(x, y);
    }
    
    while (x < 10) {
        x = x + 1;
    }
    
    for (int i = 0; i < 5; i++) {
        y = y + i;
    }
    
    return 0;
}
